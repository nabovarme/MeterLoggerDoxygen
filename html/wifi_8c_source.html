94 bool ICACHE_FLASH_ATTR acl_check_packet(struct pbuf *p) {
95  struct eth_hdr *mac_h;
96  struct ip_hdr *ip_h;
97  uint8_t proto;
98  struct udp_hdr *udp_h;
99  struct tcp_hdr *tcp_h;
100  uint16_t src_port, dest_port;
101  uint8_t *packet;
102  int i;
103 
104  if (p->len < sizeof(struct eth_hdr)) {
105  return false;
106  }
107 
108  mac_h = (struct eth_hdr *)p->payload;
109 
110  // Allow ARP
111  if (ntohs(mac_h->type) == ETHTYPE_ARP) {
112  return true;
113  }
114 
115  // Drop anything else if not IPv4
116  if (ntohs(mac_h->type) != ETHTYPE_IP) {
117  return false;
118  }
119 
120  if (p->len < sizeof(struct eth_hdr)+sizeof(struct ip_hdr)) {
121  return false;
122  }
123 
124  packet = (uint8_t*)p->payload;
125  ip_h = (struct ip_hdr *)&packet[sizeof(struct eth_hdr)];
126  proto = IPH_PROTO(ip_h);
127 
128  switch (proto) {
129  case IP_PROTO_UDP:
130  if (p->len < sizeof(struct eth_hdr)+sizeof(struct ip_hdr)+sizeof(struct udp_hdr)) {
131  return false;
132  }
133  udp_h = (struct udp_hdr *)&packet[sizeof(struct eth_hdr)+sizeof(struct ip_hdr)];
134  src_port = ntohs(udp_h->src);
135  dest_port = ntohs(udp_h->dest);
136  break;
137 
138  case IP_PROTO_TCP:
139  if (p->len < sizeof(struct eth_hdr)+sizeof(struct ip_hdr)+sizeof(struct tcp_hdr)) {
140  return false;
141  }
142  tcp_h = (struct tcp_hdr *)&packet[sizeof(struct eth_hdr)+sizeof(struct ip_hdr)];
143  src_port = ntohs(tcp_h->src);
144  dest_port = ntohs(tcp_h->dest);
145  break;
146 
147  case IP_PROTO_ICMP:
148  src_port = dest_port = 0;
149  break;
150  }
151 
152  // allow dns requests to nameserver on uplink network
153  if ((ip_h->dest.addr == dns_ip.addr) && (dest_port == 53)) {
154  return true;
155  }
156 
157  // deny connections to hosts on uplink network
158  if ((ip_h->dest.addr & sta_network_mask.addr) == sta_network_addr.addr) {
159 #ifdef DEBUG
160  os_printf("dropping packet to uplink network: src: %d.%d.%d.%d dst: %d.%d.%d.%d proto: %s sport:%d dport:%d\n",
161  IP2STR(&ip_h->src), IP2STR(&ip_h->dest),
162  proto == IP_PROTO_TCP ? "TCP" : proto == IP_PROTO_UDP ? "UDP" : "IP4", src_port, dest_port);
163 #endif
164  return false;
165  }
166 
167  // default allow everything else
168  return true;
169 }
170 #endif // AP

172 void wifi_handle_event_cb(System_Event_t *evt) {
173  uint8_t mac_str[20];
174  struct station_config stationConf;
175 
176  wifi_status = wifi_station_get_connect_status();
177  wifi_event = evt->event;
178 
179  os_memset(&stationConf, 0, sizeof(struct station_config));
180  wifi_station_get_config(&stationConf);
181 
182 #ifdef DEBUG
183 // os_printf("E%dW%d\n", evt->event, wifi_status);
184 #endif
185  switch (evt->event) {
186  case EVENT_STAMODE_CONNECTED:
187  // set default network status
188  if (os_strncmp(&stationConf.ssid, sys_cfg.sta_ssid, sizeof(sys_cfg.sta_ssid)) == 0) {
189  wifi_default_ok = true;
190  }
191  break;
192  case EVENT_STAMODE_DISCONNECTED:
193 #ifdef DEBUG
194  os_printf("disconnect from ssid %s, reason %d\n", evt->event_info.disconnected.ssid, evt->event_info.disconnected.reason);
195 #endif
196  // set default network status
197  if (os_strncmp(&stationConf.ssid, sys_cfg.sta_ssid, sizeof(sys_cfg.sta_ssid)) == 0) {
198  wifi_default_ok = false;
199  }
200  break;
201  case EVENT_STAMODE_GOT_IP:
202  // set default network status
203  if (os_strncmp(&stationConf.ssid, sys_cfg.sta_ssid, sizeof(sys_cfg.sta_ssid)) == 0) {
204  wifi_default_ok = true;
205  }
206 
207 #ifdef AP

212 
213  wifi_softap_ip_config();
214 #endif // AP

216  break;
217  case EVENT_SOFTAPMODE_STACONNECTED:
218 #ifdef DEBUG
219  os_sprintf(mac_str, MACSTR, MAC2STR(evt->event_info.sta_connected.mac));
220  os_printf("station: %s join, AID = %d\n", mac_str, evt->event_info.sta_connected.aid);
221 #endif
222 #ifdef AP
223  patch_netif_ap(my_input_ap, my_output_ap, true);
224 #endif // AP
225  break;
226  default:
227  break;
228  }

253 void ICACHE_FLASH_ATTR wifi_scan_done_cb(void *arg, STATUS status) {
254  struct bss_info *info;
255 
256  wifi_present = false;
257  wifi_fallback_present = false;
258 
259  // check if fallback network is present
260  if ((arg != NULL) && (status == OK)) {
261  info = (struct bss_info *)arg;
262  wifi_fallback_present = false;
263 
264  while (info != NULL) {
265  if ((info != NULL) && (info->ssid != NULL) && (os_strncmp(info->ssid, sys_cfg.sta_ssid, sizeof(sys_cfg.sta_ssid)) == 0)) {
266  wifi_present = true;
267  }
268  if ((info != NULL) && (info->ssid != NULL) && (os_strncmp(info->ssid, STA_FALLBACK_SSID, sizeof(STA_FALLBACK_SSID)) == 0)) {
269  wifi_fallback_present = true;
270  }
271  info = info->next.stqe_next;
272  }
273 
274  // if fallback network appeared connect to it
275  if ((wifi_fallback_present) && (!wifi_fallback_last_present)) {
276  wifi_fallback();
277  led_pattern_a();
278  }
279  // if fallback network disappeared connect to default network
280  else if ((!wifi_fallback_present) && (wifi_fallback_last_present)) {
281  wifi_default();
282  led_stop_pattern();
283  }
284 
285  wifi_fallback_last_present = wifi_fallback_present;
286 #ifdef DEBUG
287  os_printf("WiFi present: %s\n", (wifi_present ? "yes" : "no"));
288 #endif
289  }
290 
291  wifi_scan_runnning = false;
292 // os_printf("scan done\n");
293 
294  // start wifi scan timer again
295  wifi_start_scan();
296 }

108 void ICACHE_FLASH_ATTR wifi_scan_done_cb(void *arg, STATUS status) {
109  struct bss_info *info;
110 
111  wifi_present = false;
112  wifi_fallback_present = false;
113 
114  // check if fallback network is present
115  if ((arg != NULL) && (status == OK)) {
116  info = (struct bss_info *)arg;
117  wifi_fallback_present = false;
118 
119  while (info != NULL) {
120  if ((info != NULL) && (info->ssid != NULL) && (os_strncmp(info->ssid, sys_cfg.sta_ssid, sizeof(sys_cfg.sta_ssid)) == 0)) {
121  wifi_present = true;
122  }
123  if ((info != NULL) && (info->ssid != NULL) && (os_strncmp(info->ssid, STA_FALLBACK_SSID, sizeof(STA_FALLBACK_SSID)) == 0)) {
124  wifi_fallback_present = true;
125  }
126  info = info->next.stqe_next;
127  }
128  if (wifi_present == false) {
129  wifi_interrupted = true;
130  }
131 
132  // if fallback network appeared connect to it
133  if ((wifi_fallback_present) && (!wifi_fallback_last_present)) {
134  wifi_fallback();
135  led_pattern_b();
136  }
137  // if fallback network disappeared connect to default network
138  else if ((!wifi_fallback_present) && (wifi_fallback_last_present)) {
139  wifi_default();
140  led_stop_pattern();
141  }
142 
143  wifi_fallback_last_present = wifi_fallback_present;
144 #ifdef DEBUG
145  os_printf("WiFi present: %s\n", (wifi_present ? "yes" : "no"));
146  os_printf("int: %s\n", (wifi_interrupted ? "yes" : "no"));
147 #endif
148  }
149 
150  wifi_scan_runnning = false;
151 // os_printf("scan done\n");
152 
153  // start wifi scan timer again
154  os_timer_disarm(&wifi_scan_timer);
155  os_timer_setfn(&wifi_scan_timer, (os_timer_func_t *)wifi_scan_timer_func, NULL);
156  os_timer_arm(&wifi_scan_timer, WIFI_SCAN_INTERVAL, 0);
157 }

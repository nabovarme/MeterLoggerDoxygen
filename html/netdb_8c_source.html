145 /**
146  * Thread-safe variant of lwip_gethostbyname: instead of using a static
147  * buffer, this function takes buffer and errno pointers as arguments
148  * and uses these for the result.
149  *
150  * @param name the hostname to resolve
151  * @param ret pre-allocated struct where to store the result
152  * @param buf pre-allocated buffer where to store additional data
153  * @param buflen the size of buf
154  * @param result pointer to a hostent pointer that is set to ret on success
155  * and set to zero on error
156  * @param h_errnop pointer to an int where to store errors (instead of modifying
157  * the global h_errno)
158  * @return 0 on success, non-zero on error, additional error information
159  * is stored in *h_errnop instead of h_errno to be thread-safe
160  */
161 int
162 lwip_gethostbyname_r(const char *name, struct hostent *ret, char *buf,
163  size_t buflen, struct hostent **result, int *h_errnop)
164 {
165  err_t err;
166  struct gethostbyname_r_helper *h;
167  char *hostname;
168  size_t namelen;
169  int lh_errno;
170 
171  if (h_errnop == NULL) {
172  /* ensure h_errnop is never NULL */
173  h_errnop = &lh_errno;
174  }
175 
176  if (result == NULL) {
177  /* not all arguments given */
178  *h_errnop = EINVAL;
179  return -1;
180  }
181  /* first thing to do: set *result to nothing */
182  *result = NULL;
183  if ((name == NULL) || (ret == NULL) || (buf == 0)) {
184  /* not all arguments given */
185  *h_errnop = EINVAL;
186  return -1;
187  }
188 
189  namelen = strlen(name);
190  if (buflen < (sizeof(struct gethostbyname_r_helper) + namelen + 1 + (MEM_ALIGNMENT - 1))) {
191  /* buf can't hold the data needed + a copy of name */
192  *h_errnop = ERANGE;
193  return -1;
194  }
195 
196  h = (struct gethostbyname_r_helper*)LWIP_MEM_ALIGN(buf);
197  hostname = ((char*)h) + sizeof(struct gethostbyname_r_helper);
198 
199  /* query host IP address */
200  err = netconn_gethostbyname(name, &(h->addr));
201  if (err != ERR_OK) {
202  LWIP_DEBUGF(DNS_DEBUG, ("lwip_gethostbyname(%s) failed, err=%d\n", name, err));
203  *h_errnop = ENSRNOTFOUND;
204  return -1;
205  }
206 
207  /* copy the hostname into buf */
208  MEMCPY(hostname, name, namelen);
209  hostname[namelen] = 0;
210 
211  /* fill hostent */
212  h->addrs = &(h->addr);
213  h->aliases = NULL;
214  ret->h_name = (char*)hostname;
215  ret->h_aliases = &(h->aliases);
216  ret->h_addrtype = AF_INET;
217  ret->h_length = sizeof(ip_addr_t);
218  ret->h_addr_list = (char**)&(h->addrs);
219 
220  /* set result != NULL */
221  *result = ret;
222 
223  /* return success */
224  return 0;
225 }

246 /**
247  * Translates the name of a service location (for example, a host name) and/or
248  * a service name and returns a set of socket addresses and associated
249  * information to be used in creating a socket with which to address the
250  * specified service.
251  * Memory for the result is allocated internally and must be freed by calling
252  * lwip_freeaddrinfo()!
253  *
254  * Due to a limitation in dns_gethostbyname, only the first address of a
255  * host is returned.
256  * Also, service names are not supported (only port numbers)!
257  *
258  * @param nodename descriptive name or address string of the host
259  * (may be NULL -> local address)
260  * @param servname port number as string of NULL
261  * @param hints structure containing input values that set socktype and protocol
262  * @param res pointer to a pointer where to store the result (set to NULL on failure)
263  * @return 0 on success, non-zero on failure
264  */
265 int
266 lwip_getaddrinfo(const char *nodename, const char *servname,
267  const struct addrinfo *hints, struct addrinfo **res)
268 {
269  err_t err;
270  ip_addr_t addr;
271  struct addrinfo *ai;
272  struct sockaddr_in *sa = NULL;
273  int port_nr = 0;
274  size_t total_size;
275  size_t namelen = 0;
276 
277  if (res == NULL) {
278  return EAI_FAIL;
279  }
280  *res = NULL;
281  if ((nodename == NULL) && (servname == NULL)) {
282  return EAI_NONAME;
283  }
284 
285  if (servname != NULL) {
286  /* service name specified: convert to port number
287  * @todo?: currently, only ASCII integers (port numbers) are supported! */
288  port_nr = atoi(servname);
289  if ((port_nr <= 0) || (port_nr > 0xffff)) {
290  return EAI_SERVICE;
291  }
292  }
293 
294  if (nodename != NULL) {
295  /* service location specified, try to resolve */
296  err = netconn_gethostbyname(nodename, &addr);
297  if (err != ERR_OK) {
298  return EAI_FAIL;
299  }
300  } else {
301  /* service location specified, use loopback address */
302  ip_addr_set_loopback(&addr);
303  }
304 
305  total_size = sizeof(struct addrinfo) + sizeof(struct sockaddr_in);
306  if (nodename != NULL) {
307  namelen = strlen(nodename);
308  LWIP_ASSERT("namelen is too long", (namelen + 1) <= (mem_size_t)-1);
309  total_size += namelen + 1;
310  }
311  /* If this fails, please report to lwip-devel! :-) */
312  LWIP_ASSERT("total_size <= NETDB_ELEM_SIZE: please report this!",
313  total_size <= NETDB_ELEM_SIZE);
314  ai = (struct addrinfo *)memp_malloc(MEMP_NETDB);
315  if (ai == NULL) {
316  goto memerr;
317  }
318  memset(ai, 0, total_size);
319  sa = (struct sockaddr_in*)((u8_t*)ai + sizeof(struct addrinfo));
320  /* set up sockaddr */
321  inet_addr_from_ipaddr(&sa->sin_addr, &addr);
322  sa->sin_family = AF_INET;
323  sa->sin_len = sizeof(struct sockaddr_in);
324  sa->sin_port = htons((u16_t)port_nr);
325 
326  /* set up addrinfo */
327  ai->ai_family = AF_INET;
328  if (hints != NULL) {
329  /* copy socktype & protocol from hints if specified */
330  ai->ai_socktype = hints->ai_socktype;
331  ai->ai_protocol = hints->ai_protocol;
332  }
333  if (nodename != NULL) {
334  /* copy nodename to canonname if specified */
335  ai->ai_canonname = ((char*)ai + sizeof(struct addrinfo) + sizeof(struct sockaddr_in));
336  MEMCPY(ai->ai_canonname, nodename, namelen);
337  ai->ai_canonname[namelen] = 0;
338  }
339  ai->ai_addrlen = sizeof(struct sockaddr_in);
340  ai->ai_addr = (struct sockaddr*)sa;
341 
342  *res = ai;
343 
344  return 0;
345 memerr:
346  if (ai != NULL) {
347  memp_free(MEMP_NETDB, ai);
348  }
349  return EAI_MEMORY;
350 }

83 static void *rbuf_shrink_cb(void *instance, uint32_t tactic, void *env) {
84  rbuf *cur = (rbuf *)instance;
85 
86  if (tactic == 0) { /* first half */
87  return copy_rbuf_percent(cur, 0.5, 0);
88  } else if (tactic == 1) { /* second half */
89  return copy_rbuf_percent(cur, 0.5, 0.5);
90  } else if (tactic <= 18) { /* drop 1-16 bytes at start */
91  const int last_tactic = 1;
92  const size_t drop = tactic - last_tactic;
93  if (cur->size < drop) { return THEFT_DEAD_END; }
94  return copy_rbuf_subset(cur, cur->size - drop, drop);
95  } else if (tactic <= 34) { /* drop 1-16 bytes at end */
96  const int last_tactic = 18;
97  const size_t drop = tactic - last_tactic;
98  if (cur->size < drop) { return THEFT_DEAD_END; }
99  return copy_rbuf_subset(cur, cur->size - drop, 0);
100  } else if (tactic == 35) {
101  /* Divide every byte by 2, saturating at 0 */
102  rbuf *cp = copy_rbuf_percent(cur, 1, 0);

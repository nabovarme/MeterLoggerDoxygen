79 //Cgi that reads the SPI flash. Assumes 512KByte flash.
80 int ICACHE_FLASH_ATTR cgiReadFlash(HttpdConnData *connData) {
81  int *pos=(int *)&connData->cgiData;
82  if (connData->conn==NULL) {
83  //Connection aborted. Clean up.
84  return HTTPD_CGI_DONE;
85  }
86 
87  if (*pos==0) {
88  INFO("Start flash download.\n");
89  httpdStartResponse(connData, 200);
90  httpdHeader(connData, "Content-Type", "application/bin");
91  httpdEndHeaders(connData);
92  *pos=0x40200000;
93  return HTTPD_CGI_MORE;
94  }
95  //Send 1K of flash per call. We will get called again if we haven't sent 512K yet.
96  espconn_sent(connData->conn, (uint8 *)(*pos), 1024);
97  *pos+=1024;
98  if (*pos>=0x40200000+(512*1024)) return HTTPD_CGI_DONE; else return HTTPD_CGI_MORE;
99 }

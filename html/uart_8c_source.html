32 /******************************************************************************
33  * FunctionName : uart_config
34  * Description : Internal used function
35  * UART0 used for data TX/RX, RX buffer size is 0x100, interrupt enabled
36  * UART1 just used for debug output
37  * Parameters : uart_no, use UART0 or UART1 defined ahead
38  * Returns : NONE
39 *******************************************************************************/
40 LOCAL void ICACHE_FLASH_ATTR
41 uart_config(uint8 uart_no)
42 {
43  if (uart_no == UART1)
44  {
45  PIN_FUNC_SELECT(PERIPHS_IO_MUX_GPIO2_U, FUNC_U1TXD_BK);
46  }
47  else
48  {
49  /* rcv_buff size if 0x100 */
50  ETS_UART_INTR_ATTACH(uart0_rx_intr_handler, &(UartDev.rcv_buff));
51  PIN_PULLUP_DIS(PERIPHS_IO_MUX_U0TXD_U);
52  PIN_FUNC_SELECT(PERIPHS_IO_MUX_U0TXD_U, FUNC_U0TXD);
53  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTDO_U, FUNC_U0RTS);
54  }
55 
56  uart_div_modify(uart_no, UART_CLK_FREQ / (UartDev.baut_rate));
57 
58  UartDev.exist_parity = (UartDev.parity == NONE_BITS ? STICK_PARITY_DIS : STICK_PARITY_EN);
59  WRITE_PERI_REG(UART_CONF0(uart_no),
60  ((UartDev.exist_parity & UART_PARITY_EN_M) << UART_PARITY_EN_S) |
61  ((UartDev.parity & UART_PARITY_M) << UART_PARITY_S ) |
62  ((UartDev.stop_bits & UART_STOP_BIT_NUM) << UART_STOP_BIT_NUM_S) |
63  ((UartDev.data_bits & UART_BIT_NUM) << UART_BIT_NUM_S));
64 
65  //clear rx and tx fifo,not ready
66  SET_PERI_REG_MASK(UART_CONF0(uart_no), UART_RXFIFO_RST | UART_TXFIFO_RST);
67  CLEAR_PERI_REG_MASK(UART_CONF0(uart_no), UART_RXFIFO_RST | UART_TXFIFO_RST);
68 
69  //set rx fifo trigger
70 // WRITE_PERI_REG(UART_CONF1(uart_no),
71 // ((UartDev.rcv_buff.TrigLvl & UART_RXFIFO_FULL_THRHD) << UART_RXFIFO_FULL_THRHD_S) |
72 // ((96 & UART_TXFIFO_EMPTY_THRHD) << UART_TXFIFO_EMPTY_THRHD_S) |
73 // UART_RX_FLOW_EN);
74  if (uart_no == UART0)
75  {
76  //set rx fifo trigger
77  WRITE_PERI_REG(UART_CONF1(uart_no),
78  ((0x10 & UART_RXFIFO_FULL_THRHD) << UART_RXFIFO_FULL_THRHD_S) |
79  ((0x10 & UART_RX_FLOW_THRHD) << UART_RX_FLOW_THRHD_S) |
80  UART_RX_FLOW_EN |
81  (0x02 & UART_RX_TOUT_THRHD) << UART_RX_TOUT_THRHD_S |
82  UART_RX_TOUT_EN);
83  SET_PERI_REG_MASK(UART_INT_ENA(uart_no), UART_RXFIFO_TOUT_INT_ENA |
84  UART_FRM_ERR_INT_ENA);
85  }
86  else
87  {
88  WRITE_PERI_REG(UART_CONF1(uart_no),
89  ((UartDev.rcv_buff.TrigLvl & UART_RXFIFO_FULL_THRHD) << UART_RXFIFO_FULL_THRHD_S));
90  }
91 
92  //clear all interrupt
93  WRITE_PERI_REG(UART_INT_CLR(uart_no), 0xffff);
94  //enable rx_interrupt
95  SET_PERI_REG_MASK(UART_INT_ENA(uart_no), UART_RXFIFO_FULL_INT_ENA);
96 }

98 /******************************************************************************
99  * FunctionName : uart1_tx_one_char
100  * Description : Internal used function
101  * Use uart1 interface to transfer one char
102  * Parameters : uint8 TxChar - character to tx
103  * Returns : OK
104 *******************************************************************************/
105 STATUS
106 uart_tx_one_char(uint8 uart, uint8 TxChar)
107 {
108  while (true)
109  {
110  uint32 fifo_cnt = READ_PERI_REG(UART_STATUS(uart)) & (UART_TXFIFO_CNT<<UART_TXFIFO_CNT_S);
111  if ((fifo_cnt >> UART_TXFIFO_CNT_S & UART_TXFIFO_CNT) < 126) {
112  break;
113  }
114  }
115 
116  WRITE_PERI_REG(UART_FIFO(uart) , TxChar);
117  return OK;
118 }

200 LOCAL void
201 uart0_rx_intr_handler(void *para)
202 {
203  /* uart0 and uart1 intr combine togther, when interrupt occur, see reg 0x3ff20020, bit2, bit0 represents
204  * uart1 and uart0 respectively
205  */
206 // RcvMsgBuff *pRxBuff = (RcvMsgBuff *)para;
207 
208  uint8 RcvChar;
209  uint8 uart_no = UART0;//UartDev.buff_uart_no;
210 
211 
212  if (UART_FRM_ERR_INT_ST == (READ_PERI_REG(UART_INT_ST(uart_no)) & UART_FRM_ERR_INT_ST)) {
213  //os_printf("FRM_ERR\r\n");
214  WRITE_PERI_REG(UART_INT_CLR(uart_no), UART_FRM_ERR_INT_CLR);
215  }
216 
217  if (UART_RXFIFO_FULL_INT_ST == (READ_PERI_REG(UART_INT_ST(uart_no)) & UART_RXFIFO_FULL_INT_ST)) {
218  //ETS_UART_INTR_DISABLE();/////////
219  while (READ_PERI_REG(UART_STATUS(UART0)) & (UART_RXFIFO_CNT << UART_RXFIFO_CNT_S)) {
220  //WRITE_PERI_REG(0X60000914, 0x73); //WTD
221  RcvChar = READ_PERI_REG(UART_FIFO(UART0)) & 0xFF;
222 #ifdef EN61107
223  en61107_fifo_put(RcvChar);
224  if (en61107_is_eod_char(RcvChar)) { // if end of mc66 frame received
225  system_os_post(en61107_received_task_prio, 0, 0);
226  }
227 #elif defined IMPULSE
228  // nothing
229 #else
230  kmp_fifo_put(RcvChar);
231  if ((RcvChar == '\r') || (RcvChar == 0x06)) { // if end of kmp frame received or acknowledge
232  system_os_post(kmp_received_task_prio, 0, 0);
233  }
234 #endif
235  }
236  }
237  else if (UART_RXFIFO_TOUT_INT_ST == (READ_PERI_REG(UART_INT_ST(uart_no)) & UART_RXFIFO_TOUT_INT_ST)) {
238  //ETS_UART_INTR_DISABLE();/////////
239  while (READ_PERI_REG(UART_STATUS(UART0)) & (UART_RXFIFO_CNT << UART_RXFIFO_CNT_S)) {
240  //WRITE_PERI_REG(0X60000914, 0x73); //WTD
241  RcvChar = READ_PERI_REG(UART_FIFO(UART0)) & 0xFF;
242 #ifdef EN61107
243  en61107_fifo_put(RcvChar);
244  if (en61107_is_eod_char(RcvChar)) { // if end of mc66 frame received
245  system_os_post(en61107_received_task_prio, 0, 0);
246  }
247 #elif defined IMPULSE
248  // nothing
249 #else
250  kmp_fifo_put(RcvChar);
251  if ((RcvChar == '\r') || (RcvChar == 0x06)) { // if end of kmp frame received or acknowledge
252  system_os_post(kmp_received_task_prio, 0, 0);
253  }
254 #endif
255  }
256  }
257 
258  if (UART_RXFIFO_FULL_INT_ST == (READ_PERI_REG(UART_INT_ST(UART0)) & UART_RXFIFO_FULL_INT_ST)) {
259  WRITE_PERI_REG(UART_INT_CLR(UART0), UART_RXFIFO_FULL_INT_CLR);
260  }
261  else if (UART_RXFIFO_TOUT_INT_ST == (READ_PERI_REG(UART_INT_ST(UART0)) & UART_RXFIFO_TOUT_INT_ST)) {
262  WRITE_PERI_REG(UART_INT_CLR(UART0), UART_RXFIFO_TOUT_INT_CLR);
263  }
264  //ETS_UART_INTR_ENABLE();
265 }

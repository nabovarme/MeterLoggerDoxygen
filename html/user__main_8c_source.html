633 #ifdef IMPULSE
634 void gpio_int_handler(uint32_t interrupt_mask, void *arg) {
635  uint32_t gpio_status;
636  bool impulse_pin_state;
637 
638  uint32_t impulse_edge_to_edge_time;
639 
640  gpio_intr_ack(interrupt_mask);
641 
642  ETS_GPIO_INTR_DISABLE(); // Disable gpio interrupts
643  gpio_pin_intr_state_set(GPIO_ID_PIN(5), GPIO_PIN_INTR_DISABLE);
644  //wdt_feed();
645 
646  gpio_status = GPIO_REG_READ(GPIO_STATUS_ADDRESS);
647  //clear interrupt status
648  GPIO_REG_WRITE(GPIO_STATUS_W1TC_ADDRESS, gpio_status);
649 
650  os_delay_us(1000); // wait 1 mS to avoid reading on slope
651  impulse_pin_state = GPIO_REG_READ(GPIO_IN_ADDRESS) & BIT5;
652  if (impulse_pin_state) { // rising edge
653  impulse_rising_edge_time = system_get_time();
654 
655  if (impulse_rising_edge_time > impulse_falling_edge_time) {
656  impulse_edge_to_edge_time = impulse_rising_edge_time - impulse_falling_edge_time;
657  }
658  else {
659  // system time wrapped
660 #ifdef DEBUG
661  printf("wrapped\n");
662 #endif
663  impulse_edge_to_edge_time = UINT32_MAX - impulse_falling_edge_time + impulse_rising_edge_time;
664  }
665 
666  // check if impulse period is 100 mS...
667 #ifdef DEBUG
668  printf("imp: %uuS\n", impulse_rising_edge_time - impulse_falling_edge_time);
669 #endif // DEBUG
670  if ((impulse_edge_to_edge_time > 80 * 1000) && (impulse_edge_to_edge_time < 120 * 1000)) {
671  // arm the debounce timer to enable GPIO interrupt again
672  sys_cfg.impulse_meter_count++;
673  os_timer_disarm(&impulse_meter_calculate_timer);
674  os_timer_setfn(&impulse_meter_calculate_timer, (os_timer_func_t *)impulse_meter_calculate_timer_func, NULL);
675  os_timer_arm(&impulse_meter_calculate_timer, 100, 0);
676  }
677  }
678  else { // falling edge
679  impulse_falling_edge_time = system_get_time();
680  }
681 
682  // enable gpio interrupt again
683  gpio_pin_intr_state_set(GPIO_ID_PIN(5), GPIO_PIN_INTR_ANYEDGE); // Interrupt on falling GPIO4 edge
684  ETS_GPIO_INTR_ENABLE();
685 }

117 //Decode a percent-encoded value.
118 //Takes the valLen bytes stored in val, and converts it into at most retLen bytes that
119 //are stored in the ret buffer. Returns the actual amount of bytes used in ret. Also
120 //zero-terminates the ret buffer.
121 int httpdUrlDecode(char *val, int valLen, char *ret, int retLen) {
122  int s=0, d=0;
123  int esced=0, escVal=0;
124  while (s<valLen && d<retLen) {
125  if (esced==1) {
126  escVal=httpdHexVal(val[s])<<4;
127  esced=2;
128  } else if (esced==2) {
129  escVal+=httpdHexVal(val[s]);
130  ret[d++]=escVal;
131  esced=0;
132  } else if (val[s]=='%') {
133  esced=1;
134  } else if (val[s]=='+') {
135  ret[d++]=' ';
136  } else {
137  ret[d++]=val[s];
138  }
139  s++;
140  }
141  if (d<retLen) ret[d]=0;
142  return d;
143 }

145 //Find a specific arg in a string of get- or post-data.
146 //Line is the string of post/get-data, arg is the name of the value to find. The
147 //zero-terminated result is written in buff, with at most buffLen bytes used. The
148 //function returns the length of the result, or -1 if the value wasn't found. The
149 //returned string will be urldecoded already.
150 ICACHE_FLASH_ATTR
151 int httpdFindArg(char *line, char *arg, char *buff, int buffLen) {
152  char *p, *e;
153  if (line==NULL) return 0;
154  p=line;
155  while(p!=NULL && *p!='\n' && *p!='\r' && *p!=0) {
156  INFO("findArg: %s\n", p);
157  if (os_strncmp(p, arg, os_strlen(arg))==0 && p[strlen(arg)]=='=') {
158  p+=os_strlen(arg)+1; //move p to start of value
159  e=(char*)os_strstr(p, "&");
160  if (e==NULL) e=p+os_strlen(p);
161  INFO("findArg: val %s len %d\n", p, (e-p));
162  return httpdUrlDecode(p, (e-p), buff, buffLen);
163  }
164  p=(char*)os_strstr(p, "&");
165  if (p!=NULL) p+=1;
166  }
167  INFO("Finding %s in %s: Not found :/\n", arg, line);
168  return -1; //not found
169 }

335  int i;

338  char *e;
339 
340  //Skip past the space after POST/GET
341  i=0;
342  while (h[i]!=' ') i++;

344

350  INFO("URL = %s\n", conn->url);

352  conn->getArgs=(char*)os_strstr(conn->url, "?");
353  if (conn->getArgs!=0) {
354  *conn->getArgs=0;
355  conn->getArgs++;
356  INFO("GET args = %s\n", conn->getArgs);
357  } else {
358  conn->getArgs=NULL;
359  }

371  conn->priv->postPos=0;

376 //Callback called when there's data available on a socket.
377 ICACHE_FLASH_ATTR
378 static void httpdRecvCb(void *arg, char *data, unsigned short len) {
379  int x;
380  char *p, *e;
381  char sendBuff[MAX_SENDBUFF_LEN];
382  HttpdConnData *conn=httpdFindConnData(arg);
383  if (conn==NULL) return;
384  conn->priv->sendBuff=sendBuff;
385  conn->priv->sendBuffLen=0;
386 
387  for (x=0; x<len; x++) {
388  if (conn->postLen<0) {
389  //This byte is a header byte.
390  if (conn->priv->headPos!=MAX_HEAD_LEN) conn->priv->head[conn->priv->headPos++]=data[x];
391  conn->priv->head[conn->priv->headPos]=0;
392  //Scan for /r/n/r/n
393  if (data[x]=='\n' && (char *)os_strstr(conn->priv->head, "\r\n\r\n")!=NULL) {
394  //Indicate we're done with the headers.
395  conn->postLen=0;
396  //Reset url data
397  conn->url=NULL;
398  //Find end of next header line
399  p=conn->priv->head;
400  while(p<(&conn->priv->head[conn->priv->headPos-4])) {
401  e=(char *)os_strstr(p, "\r\n");
402  if (e==NULL) break;
403  e[0]=0;
404  httpdParseHeader(p, conn);
405  p=e+2;
406  }
407  //If we don't need to receive post data, we can send the response now.
408  if (conn->postLen==0) {
409  httpdSendResp(conn);
410  }
411  }
412  } else if (conn->priv->postPos!=-1 && conn->postLen!=0 && conn->priv->postPos <= conn->postLen) {
413  //This byte is a POST byte.
414  conn->postBuff[conn->priv->postPos++]=data[x];
415  if (conn->priv->postPos>=conn->postLen) {
416  //Received post stuff.
417  conn->postBuff[conn->priv->postPos]=0; //zero-terminate
418  conn->priv->postPos=-1;
419  INFO("Post data: %s\n", conn->postBuff);
420  //Send the response.
421  httpdSendResp(conn);
422  break;
423  }
424  }
425  }
426  xmitSendBuff(conn);
427 }

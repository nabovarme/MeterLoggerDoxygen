35  if (r && strncmp(hdr, "Basic", 5)==0) {
36  r=base64_decode(strlen(hdr)-6, hdr+6, sizeof(userpass), (unsigned char *)userpass);
37  if (r<0) r=0; //just clean out string on decode error
38  userpass[r]=0; //zero-terminate user:pass string
39 // os_printf("Auth: %s\n", userpass);
40  while (((AuthGetUserPw)(connData->cgiArg))(connData, no,
41  user, AUTH_MAX_USER_LEN, pass, AUTH_MAX_PASS_LEN)) {
42  //Check user/pass against auth header
43  if (strlen(userpass)==strlen(user)+strlen(pass)+1 &&
44  os_strncmp(userpass, user, strlen(user))==0 &&
45  userpass[strlen(user)]==':' &&
46  os_strcmp(userpass+strlen(user)+1, pass)==0) {

49  }

51  }
52  }
53

55  httpdStartResponse(connData, 401);

58  httpdEndHeaders(connData);

60  //Okay, all done.

62 }

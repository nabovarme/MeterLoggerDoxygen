216 lwip_sanity_check(void)
217 {
218  /* Warnings */
219 #if LWIP_NETCONN
220  if (MEMP_NUM_NETCONN > (MEMP_NUM_TCP_PCB+MEMP_NUM_TCP_PCB_LISTEN+MEMP_NUM_UDP_PCB+MEMP_NUM_RAW_PCB))
221  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: MEMP_NUM_NETCONN should be less than the sum of MEMP_NUM_{TCP,RAW,UDP}_PCB+MEMP_NUM_TCP_PCB_LISTEN\n"));
222 #endif /* LWIP_NETCONN */
223 #if LWIP_TCP
224  if (MEMP_NUM_TCP_SEG < TCP_SND_QUEUELEN)
225  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: MEMP_NUM_TCP_SEG should be at least as big as TCP_SND_QUEUELEN\n"));
226  if (TCP_SND_BUF < 2 * TCP_MSS)
227  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: TCP_SND_BUF must be at least as much as (2 * TCP_MSS) for things to work smoothly\n"));
228  if (TCP_SND_QUEUELEN < (2 * (TCP_SND_BUF/TCP_MSS)))
229  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: TCP_SND_QUEUELEN must be at least as much as (2 * TCP_SND_BUF/TCP_MSS) for things to work\n"));
230  if (TCP_SNDLOWAT >= TCP_SND_BUF)
231  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: TCP_SNDLOWAT must be less than TCP_SND_BUF.\n"));
232  if (TCP_SNDQUEUELOWAT >= TCP_SND_QUEUELEN)
233  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: TCP_SNDQUEUELOWAT must be less than TCP_SND_QUEUELEN.\n"));
234  if (TCP_WND > (PBUF_POOL_SIZE*PBUF_POOL_BUFSIZE))
235  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: TCP_WND is larger than space provided by PBUF_POOL_SIZE*PBUF_POOL_BUFSIZE\n"));
236  if (TCP_WND < TCP_MSS)
237  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: TCP_WND is smaller than MSS\n"));
238 #endif /* LWIP_TCP */
239 #if LWIP_SOCKET
240  /* Check that the SO_* socket options and SOF_* lwIP-internal flags match */
241  if (SO_ACCEPTCONN != SOF_ACCEPTCONN)
242  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: SO_ACCEPTCONN != SOF_ACCEPTCONN\n"));
243  if (SO_REUSEADDR != SOF_REUSEADDR)
244  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: SO_REUSEADDR != SOF_REUSEADDR\n"));
245  if (SO_KEEPALIVE != SOF_KEEPALIVE)
246  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: SO_KEEPALIVE != SOF_KEEPALIVE\n"));
247  if (SO_BROADCAST != SOF_BROADCAST)
248  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: SO_BROADCAST != SOF_BROADCAST\n"));
249  if (SO_LINGER != SOF_LINGER)
250  LWIP_PLATFORM_DIAG(("lwip_sanity_check: WARNING: SO_LINGER != SOF_LINGER\n"));
251 #endif /* LWIP_SOCKET */
252 }
